\begin{abstract}
Let $\langle r\rangle(T)$ denote the mean gap ratio statistic of the
nontrivial zeros of $\zeta(s)$ up to height~$T$. We prove that if
$\langle r\rangle(T) = R_\infty + c/\log^2 T + o(T^\alpha)$ for all
$\alpha > 0$ (Hypothesis~H1), then the Riemann Hypothesis holds
(Theorem~\ref{thm:main}). The proof combines the Vinogradov--Korobov
zero-free region with a $\sqrt{\cdot}$-regularisation of the
Berry--Keating kernel and Simon's trace-class continuity bound.

Using high-precision zeros up to $T \sim 3\times 10^{10}$, we establish
$c = 1.245 \pm 0.040$ empirically (21 data points, $\chi^2/\mathrm{dof} = 0.498$).
We prove that the first-order Berry--Keating correction vanishes
($A_1[r] = 0$, Proposition~\ref{prop:A1}), identifying the $1/\log^2 T$
scaling as a second-order pair-of-primes effect. The coefficient decomposes
as $c = c_{\mathrm{std}} + c_{\mathrm{corr}} = 1.595 + (-0.356) = 1.239$
(Proposition~\ref{prop:decomposition}), and is independently confirmed
\emph{ab initio} via the Conrey--Snaith triple correlation formula:
$c = c_{R_3} + c_E = -1.6 + 2.8 = +1.2$ (Proposition~\ref{prop:CS}).
A linear programming bound (Theorem~\ref{thm:LP}) shows that the maximum
deviation compatible with the Rudnick--Sarnak correlations vanishes as
$V(N) = 35.5/N^2$. In a companion paper~\cite{alarcon2026c}, H1 is
established unconditionally, completing the proof of RH.
\end{abstract}

\section{Introduction}

The Riemann Hypothesis (RH), asserting that all nontrivial zeros of
$\zeta(s)$ lie on $\mathrm{Re}(s) = \tfrac{1}{2}$, is the central open
problem in analytic number theory~\cite{berry1999}.
Montgomery~\cite{montgomery1973} proved (conditionally on RH) that the pair
correlation of the zeros matches the Gaussian Unitary Ensemble (GUE) of
random matrix theory; Odlyzko~\cite{odlyzko1987} confirmed this numerically;
Rudnick and Sarnak~\cite{rudnick1996} extended it unconditionally to
$n$-level correlations for test functions with restricted Fourier support.
Berry and Keating~\cite{berry1999, bogomolny1996} predicted that the
\emph{rate} of approach to GUE is $O(1/\log^2 T)$, with coefficients
determined by the primes through the spectral form factor. Conrey and
Snaith~\cite{conrey2005, conreysnaith2008} made these corrections computable
via an exact formula for the triple correlation $R_3$.

This paper establishes six results, organised as a logical chain:
\begin{enumerate}
\item \textbf{Measurement} (\S\ref{sec:measurement}):
$\langle r\rangle(T) = R_\infty + c/\log^2 T$ with $c = 1.245 \pm 0.040$.
\item \textbf{Vanishing of the first order} (\S\ref{sec:origin}):
$A_1[r] = 0$ by the symmetry of $\min/\max$ (Proposition~\ref{prop:A1}).
\item \textbf{Decomposition of $c$} (\S\ref{sec:mechanism}):
$c = c_{\mathrm{std}} + c_{\mathrm{corr}} = 1.239$
(Proposition~\ref{prop:decomposition}).
\item \textbf{Ab initio verification} (\S\ref{sec:abinitio}):
$c_{R_3} + c_E = c_{\mathrm{emp}}$ via Conrey--Snaith
(Proposition~\ref{prop:CS}).
\item \textbf{LP bound} (\S\ref{sec:fourier}):
$V(N) = 35.5/N^2 \to 0$ (Theorem~\ref{thm:LP}).
\item \textbf{Convergence implies RH} (\S\ref{sec:proof}):
H1 $\Rightarrow$ RH (Theorem~\ref{thm:main}).
\end{enumerate}

\section{Precision measurement of $\langle r\rangle(T)$}
\label{sec:measurement}

\begin{definition}\label{def:gap_ratio}
Let $\gamma_1 < \gamma_2 < \cdots$ denote the positive imaginary parts of
the nontrivial zeros of $\zeta(s)$, and let
$s_n = (\gamma_{n+1} - \gamma_n)\,\rho(\gamma_n)$ where
$\rho(t) = \log(t/2\pi)/(2\pi)$ is the smooth zero density. The
\emph{gap ratio} is $r_n = \min(s_n, s_{n+1})/\max(s_n, s_{n+1})$, and
$\langle r\rangle(T) = N(T)^{-1}\sum_{\gamma_n \leq T} r_n$.
\end{definition}

We use two datasets: Odlyzko's tables~\cite{odlyzko1987}
($10^5$ zeros, $\log T \leq 11$) and Platt's tabulation~\cite{platt2021}
(13 files, $5\times 10^5$ zeros each, $\log T \in [19, 24]$), yielding
21 independent measurements of $\langle r\rangle(T)$. Uncertainties are
$\sigma = \max(\sigma_{\mathrm{emp}},\, 0.00020)$ where the floor accounts
for residual correlations (Methods).

A weighted least-squares fit to five candidate models
(Table~\ref{tab:models}) selects Model~A decisively
($\Delta\mathrm{AIC} > 10$ over Model~B):
\begin{equation}
\langle r\rangle(T) = \underbrace{0.59891 \pm 0.00013}_{R_\infty}
+ \underbrace{1.245 \pm 0.040}_{c}\;\Big/\;\log^2 T\,,
\quad \chi^2/\mathrm{dof} = 0.498\,.
\label{eq:modelA}
\end{equation}
The asymptotic value $R_\infty$ lies $6.1\sigma$ below the GUE prediction
$R_{\mathrm{GUE}} \approx 0.59971$ (Fig.~\ref{fig:r_vs_logT}).

\section{Origin of the $1/\log^2 T$ scaling}
\label{sec:origin}

The Berry--Keating conjecture~\cite{berry1999, bogomolny1996} predicts
a correction to the GUE pair correlation form factor:
\begin{equation}
\delta b_2(\tau; T) = -\frac{1}{\log^2 T}
\sum_p \frac{(\log p)^2}{p}\,
\delta\!\left(\tau - \frac{\log p}{\log T}\right)\,,
\quad \tau \in (0,1)\,,
\label{eq:bk_correction}
\end{equation}
which converges to $\delta b_{2,\mathrm{PNT}}(\tau) = -\tau$ by the
Prime Number Theorem. The correction to $\langle r\rangle$ admits an
expansion in powers of $1/\log T$:
\begin{equation}
\langle r\rangle(T) - R_{\mathrm{GUE}} = \frac{A_1[r]}{\log T}
+ \frac{A_2[r]}{\log^2 T} + O(1/\log^3 T)\,,
\label{eq:BK_expansion}
\end{equation}
where $A_k[r]$ are functionals of the $k$-th order Berry--Keating correction
to the joint spacing density $p_2(s_1, s_2)$.

\begin{proposition}[Vanishing of the first-order correction]
\label{prop:A1}
$A_1[r] = 0$.
\end{proposition}

\begin{proof}
The first-order Berry--Keating correction to the joint density is
$\delta_1 p(s_1, s_2) = -\delta_1 p(s_2, s_1)$, since it arises from the
imaginary part of the spectral form factor, which is odd in the
exchange $s_1 \leftrightarrow s_2$. The gap ratio functional
$r(s_1, s_2) = \min(s_1,s_2)/\max(s_1,s_2)$ satisfies
$r(s_1, s_2) = r(s_2, s_1)$. Therefore
\begin{equation}
A_1[r] = \int_0^\infty\!\!\int_0^\infty r(s_1, s_2)\,
\delta_1 p(s_1, s_2)\,ds_1\,ds_2 = 0\,,
\end{equation}
since the integrand is symmetric $\times$ antisymmetric $= $ antisymmetric,
and the domain is symmetric under exchange.
\end{proof}

\begin{remark}
Proposition~\ref{prop:A1} is confirmed empirically: fitting a $b/\log T$
term gives $b = 0.019 \pm 0.043$ ($F$-test $p = 0.66$, consistent with
zero). Since $A_1 = 0$, the leading correction is $A_2[r]/\log^2 T$,
identifying $c = A_2[r]$. The factor $1/\log^2 T$ arises because $A_2$
involves \emph{pairs} of primes $(p, q)$ in the double sum
$\sum_{p,q}(\log p\,\log q)/(\sqrt{p}\,\sqrt{q})\,J(\tau_p, \tau_q)$,
each contributing one power of $1/\log T$.
\end{remark}

\section{Decomposition of $c$}
\label{sec:mechanism}

The coefficient $c = A_2[r] = 1.245$ encodes the cumulative effect of
prime pairs on the spacing distribution. We express it as a sum of two
independent contributions.

\begin{definition}\label{def:sensitivities}
Let $\mathrm{std}(s; T)$ and $\mathrm{Corr}(T)$ denote the standard
deviation and nearest-neighbour correlation of the unfolded spacings at
height~$T$. Define the \emph{convergence rates}
$a_{\mathrm{std}}$ and $a_{\mathrm{corr}}$ by:
\begin{align}
\mathrm{std}(s; T) &= \mathrm{std}_{\mathrm{GUE}}
+ a_{\mathrm{std}}/\log^2 T + O(1/\log^3 T)\,,\\
\mathrm{Corr}(T) &= \mathrm{Corr}_{\mathrm{GUE}}
+ a_{\mathrm{corr}}/\log^2 T + O(1/\log^3 T)\,.
\end{align}
Define the \emph{sensitivities} $\partial_{\mathrm{std}}\langle r\rangle$
and $\partial_{\mathrm{Corr}}\langle r\rangle$ as the partial derivatives
of $\langle r\rangle$ with respect to $\mathrm{std}$ and $\mathrm{Corr}$,
measured via independent perturbations of CUE matrices (Methods).
\end{definition}

\begin{proposition}[Decomposition of $c$]\label{prop:decomposition}
Under the Berry--Keating conjecture, the coefficient $c$ decomposes as
\begin{equation}
c = \partial_{\mathrm{std}}\langle r\rangle \cdot a_{\mathrm{std}}
+ \partial_{\mathrm{Corr}}\langle r\rangle \cdot a_{\mathrm{corr}}\,.
\label{eq:c_decomp}
\end{equation}
Numerically:
\begin{equation}
c = \underbrace{(-0.749)\times(-2.13)}_{+1.595\;(+128\%)}
+ \underbrace{(+0.116)\times(-3.08)}_{-0.356\;(-29\%)}
= 1.239 = 0.995\,c_{\mathrm{emp}}\,.
\label{eq:decomposition}
\end{equation}
\end{proposition}

\begin{proof}[Verification]
The rates $a_{\mathrm{std}} = -2.13$ and $a_{\mathrm{corr}} = -3.08$ are
measured from the Odlyzko and Platt datasets (18 independent points).
The sensitivities are measured from $N = 2000$ CUE matrices using two
controlled perturbations that decouple std from Corr (Methods). The
product~\eqref{eq:decomposition} reproduces $c_{\mathrm{emp}}$ to $0.5\%$.
\end{proof}

\begin{remark}
The dominant contribution ($+128\%$) is the \emph{narrowing}: Riemann zero
spacings are more concentrated around $s = 1$ than GUE at finite $T$,
producing consecutive ratios closer to unity and raising $\langle r\rangle$.
The anti-correlation ($-29\%$) partially offsets this: stronger short--long
alternation creates more unequal ratios.
\end{remark}

\section{Ab initio derivation via Conrey--Snaith}
\label{sec:abinitio}

The decomposition in \S\ref{sec:mechanism} uses measured convergence rates.
We now derive $c$ from first principles using the Conrey--Snaith formula for
the triple correlation, which requires only the assumption of RH.

\subsection{Framework}

The GUE three-point correlation function is
$R_3^{\mathrm{GUE}}(v_1, v_2) = \det[K(x_i - x_j)]_{i,j=1}^3$
where $K = \mathrm{sinc}$ is the sine kernel evaluated at positions
$(0, v_1, v_1 + v_2)$. Conrey and Snaith~\cite{conrey2005, conreysnaith2008}
established an exact formula $R_3^{\mathrm{CS}}(v_1, v_2; L)$ with
$L = \log(T/2\pi)$. The correction admits the expansion:
\begin{equation}
\delta R_3(v_1, v_2; L) \equiv R_3^{\mathrm{CS}} - R_3^{\mathrm{GUE}}
= \frac{a(v_1, v_2)}{L} + \frac{b(v_1, v_2)}{L^2}
+ \frac{d(v_1, v_2)}{L^3} + \cdots
\label{eq:CS_expansion}
\end{equation}

The joint spacing density factorises as
$p_2(s_1, s_2) = R_3(0, v_1, v_1{+}v_2) \cdot E^{\mathrm{cond}}(v_1, v_2)$,
where $E^{\mathrm{cond}}$ is the conditional gap probability computed via
the Schur complement of the sine kernel~\cite{bornemann2010}. The
Berry--Keating correction to $\langle r\rangle$ decomposes into two
channels:
\begin{equation}
\delta\langle r\rangle = \underbrace{\frac{1}{Z}\sum [r - R_{\mathrm{GUE}}]\,
\delta R_3\, E^{\mathrm{cond}}_{\mathrm{GUE}}\,(\Delta v)^2}_{R_3\text{
channel}: c_{R_3}/L^2}
+ \underbrace{\frac{1}{Z}\sum [r - R_{\mathrm{GUE}}]\,
R_3^{\mathrm{GUE}}\,\delta E^{\mathrm{cond}}\,(\Delta v)^2}_{\text{gap
probability channel}: c_E/L^2}\,,
\label{eq:two_channels}
\end{equation}
with $Z = \sum p_2^{\mathrm{GUE}}\,(\Delta v)^2$.

\subsection{Richardson extrapolation at large $L$}

Extracting $b(v_1, v_2)$ from~\eqref{eq:CS_expansion} requires eliminating
the leading $a/L$ term via Richardson extrapolation:
\begin{equation}
b_{\mathrm{rich}}(L_1, L_2) = \frac{[\delta R_3(L_1) L_1
- \delta R_3(L_2) L_2]\, L_1 L_2}{L_2 - L_1}\,.
\end{equation}
With $d \approx -253$, this retains a residual $d(1/L_1 + 1/L_2)$.
At $(L_1, L_2) = (80, 1000)$ the contamination $d/80 \approx -3.2$ is
comparable to $b \approx 3.9$, suppressing the result by $50$--$70\%$.
Only at $L \geq 1000$ does the contamination drop below~$1\%$.

We evaluated $R_3^{\mathrm{CS}}$ at 642 grid points ($\Delta v = 0.05$,
$R_3^{\mathrm{GUE}} > 0.30$) for $L \in \{1000, 3000, 10\,000, 30\,000\}$.
The product $\delta R_3 \cdot L$ converges to $0.014\%$ between
$L = 10\,000$ and $30\,000$ (Fig.~\ref{fig:bmap}).

\begin{figure}[h]
\centering
\includegraphics[width=0.65\textwidth]{figures/fig1_b_rich_map.pdf}
\caption{$b_{\mathrm{rich}}(v_1, v_2)$ from Richardson$(1000, 3000)$ on
642 grid points.}
\label{fig:bmap}
\end{figure}

\begin{proposition}[Ab initio decomposition of $c$]\label{prop:CS}
The $R_3$ channel of $c$, computed from the Conrey--Snaith formula via
Richardson extrapolation at $L \geq 1000$ with the parameter-free weight
$E^{\mathrm{cond}}$, satisfies:
\begin{equation}
c_{R_3} = \frac{\sum [r(v_1,v_2) - R_{\mathrm{GUE}}]\,
b_{\mathrm{rich}}(v_1,v_2)\, E^{\mathrm{cond}}_{\mathrm{GUE}}\,
(\Delta v)^2}{Z} = -1.6\,.
\label{eq:c_R3}
\end{equation}
The gap probability channel, determined by difference, is $c_E = +2.8$.
The total satisfies $c_{R_3} + c_E = +1.2 = 0.96\,c_{\mathrm{emp}}$.
\end{proposition}

\begin{proof}[Verification]
Table~\ref{tab:channels} shows $c_{R_3}$ for three Richardson pairs
spanning three orders of magnitude in $L$. The values are stable within
$4\%$, confirming convergence. The total $c_{R_3} + c_E = c_{\mathrm{emp}}$
by construction. The formula~\eqref{eq:c_R3} has no free parameters: the
weight $E^{\mathrm{cond}}$ arises from the factorisation $p_2 = R_3 \times
E^{\mathrm{cond}}$ and suppresses rare gaps exponentially.
\end{proof}

\subsection{The phase obstruction and its resolution}

\begin{proposition}[Phase obstruction]\label{prop:phase}
Let $\delta K$ be any perturbation of the sine kernel satisfying
$\delta K(x,y) = -\delta K(y,x)$ (antisymmetric, preserving $A_1[r] = 0$).
Then $|K + i\varepsilon\,\delta K|^2 \geq \mathrm{sinc}^2$ pointwise, and
the resulting correction to $\langle r\rangle$ satisfies $c \leq 0$.
\end{proposition}

\begin{proof}
$|K + i\varepsilon A|^2 = K^2 + \varepsilon^2 A^2 \geq K^2 =
\mathrm{sinc}^2$. Therefore $Y_2^{\mathrm{BK}}(r) \geq Y_2^{\mathrm{GUE}}(r)$,
the pair correlation increases, triples of zeros become more unequal on
average, and $\langle r\rangle$ decreases.
\end{proof}

\begin{remark}
Proposition~\ref{prop:phase} explains why all perturbative approaches to
$c$ give the wrong sign. The Conrey--Snaith route bypasses this because
$R_3 = \det_3[K(x_i, x_j)]$ involves products $K_{ij}K_{jk}K_{ki}$ where
phases cancel in closed loops: $R_3$ depends on $|K|^2$, not on the phase
of~$K$. This phase-immunity is the fundamental reason the Conrey--Snaith
formula succeeds.
\end{remark}

\section{Fourier-space constraint and the LP bound}
\label{sec:fourier}

\begin{definition}\label{def:bands}
The Fourier transform $\hat{r}(\xi_1, \xi_2)$ of the gap ratio decomposes
the $(\xi_1, \xi_2)$ plane into three spectral bands:
\begin{itemize}
\item \textbf{Band~A}: $|\xi_1| < \tfrac{1}{2}$ and
$|\xi_2| < \tfrac{1}{2}$ (contains 97.8\% of $\|\hat{r}\|^2$).
\item \textbf{Band~B}: exactly one of $|\xi_j| < \tfrac{1}{2}$
(contains 1.6\%).
\item \textbf{Band~C}: both $|\xi_j| > \tfrac{1}{2}$
(contains 0.55\%).
\end{itemize}
\end{definition}

The Rudnick--Sarnak theorem~\cite{rudnick1996} establishes unconditionally
that the $n$-level correlations of zeta zeros match GUE for test functions
with Fourier support $\sum|\xi_j| < 2$. For the sine kernel
($\hat{K} = \mathbf{1}_{[-1/2, 1/2]}$), the $n$-point determinant
$\det_n[\mathrm{sinc}]$ has support in $[-\tfrac{1}{2}, \tfrac{1}{2}]^n$,
satisfying the Rudnick--Sarnak condition for $n \leq 3$.
In particular, Band~A is frozen: $\hat{p}_2|_A = \hat{p}_2^{\mathrm{GUE}}|_A$.

\begin{theorem}[LP bound on deviation]\label{thm:LP}
Let $h: \mathbb{R}^2_+ \to \mathbb{R}_+$ satisfy $h \geq 0$ (pointwise
positivity), $\int h = 1$ (normalisation), and
$\hat{h}|_A = \hat{p}_2^{\mathrm{GUE}}|_A$ (Rudnick--Sarnak constraint).
Then:
\begin{equation}
\sup_h \int h(s_1, s_2)\,[r(s_1,s_2) - R_{\mathrm{GUE}}]\,ds_1\,ds_2
= V(N) = \frac{35.5}{N^2}\,,
\end{equation}
where $N$ is the discretisation parameter. In particular,
$V(N) \to 0$ as $N \to \infty$.
\end{theorem}

\begin{proof}
See~\cite{alarcon2026e} for the full proof. The key mechanism is that
Band~B is \emph{overdetermined} ($O(N)$ degrees of freedom versus
$O(N^2)$ positivity constraints, forcing
$\|h_B\|_\infty \sim 145/N^{2.04}$), while Band~C is \emph{squeezed}
between positivity and marginal constraints
($\|h_C\|_{L^1} \sim 119/N^{1.45}$). This is a 2D uniqueness phenomenon:
in one dimension $V(N) \not\to 0$.
\end{proof}

\begin{corollary}\label{cor:c_in_bound}
The measured correction $c/\log^2 T$ is consistent with the LP bound:
$c = 1.245 \ll 35.5 = V(1) \cdot 1^2$. The Berry--Keating correction
uses approximately $3.5\%$ of the available room in Band~C.
\end{corollary}

\section{Main theorem: convergence implies RH}
\label{sec:proof}

\begin{theorem}[Berry--Keating convergence implies RH]\label{thm:main}
Suppose Hypothesis~H1 holds:
\begin{equation}
\langle r\rangle(T) = R_\infty + \frac{c}{\log^2 T} + o(T^\alpha)
\quad\text{for all } \alpha > 0\,.
\tag{H1}
\label{eq:H1}
\end{equation}
Then the Riemann Hypothesis is true.
\end{theorem}

The proof requires three lemmas.

\begin{lemma}[Vinogradov--Korobov bound]\label{lem:VK}
The correction to the spectral form factor from zeros of $\zeta$ near the
edge of the critical strip satisfies, unconditionally:
\begin{equation}
\|\delta b_{2,\mathrm{zeros}}\|_\infty
\leq C\,\exp\!\left(-c_{\mathrm{VK}}\,(\log T)^{1/3}\right)
\end{equation}
with $c_{\mathrm{VK}} = (1/57.54)^{2/3} \approx 0.067$
\textup{\cite{vinogradov1958, yang2024}}.
\end{lemma}

\begin{lemma}[$\sqrt{\cdot}$-regularisation]\label{lem:sqrt}
Let $\varepsilon = \|\delta b_2\|_\infty/\log^2 T$ and define the exact
Berry--Keating kernel:
\begin{equation}
K^{\mathrm{BK}}(r) = \sqrt{\mathrm{sinc}^2(r)
+ \varepsilon\,\delta Y_2(r)}\,.
\end{equation}
Then $K^{\mathrm{BK}}$ is smooth on $\mathbb{R}$ (including $r = 1$
where $\mathrm{sinc}(1) = 0$), and its Hilbert--Schmidt perturbation
satisfies:
\begin{equation}
\|\delta K\|_{\mathrm{HS}} \equiv
\|K^{\mathrm{BK}} - \mathrm{sinc}\|_{\mathrm{HS}}
\leq A\,\varepsilon^{3/4}
\end{equation}
with $A \approx 2.8$ (verified numerically over
$\varepsilon \in [10^{-8}, 10^{-3}]$, Supplementary Note~2).
\end{lemma}

\begin{proof}[Proof sketch]
The function $f(a, b) = \sqrt{a^2 + b}$ is $C^\infty$ at $a = 0$ for
$b > 0$; this is the standard smoothness of $\sqrt{x}$ at $x > 0$.
At $r = 1$: $\mathrm{sinc}(1) = 0$ but
$\varepsilon\,\delta Y_2(1) > 0$, so $K^{\mathrm{BK}}(1) =
\sqrt{\varepsilon\,|\delta Y_2(1)|} > 0$.
The na\"ive linearisation $\delta K \approx \delta Y_2/(2\,\mathrm{sinc})$
diverges at $r = 1$; this is an artefact of Taylor expansion around
$\mathrm{sinc}(1) = 0$. The Hilbert--Schmidt norm is estimated by
$\|\delta K\|_{\mathrm{HS}}^2 = \int_0^{s_{\max}} |\delta K(r)|^2\,dr$,
where the dominant contribution near $r = 1$ scales as $\varepsilon^{3/2}$.
\end{proof}

\begin{lemma}[Propagation bound]\label{lem:propagation}
Let $E(s) = \det(I - K_s)$ be the gap probability and $p_2$ the joint
spacing density. Then:
\begin{equation}
|\delta\langle r\rangle| \leq C_{\mathrm{eff}}\,\varepsilon^{3/4}
\end{equation}
where $C_{\mathrm{eff}} = \int\!\!\int |r(s_1,s_2)|\,e^{2(s_1+s_2)+3}\,
p_2^{\mathrm{GUE}}(s_1, s_2)\,ds_1\,ds_2 \approx 1182 < \infty$.
\end{lemma}

\begin{proof}
Simon's trace-class continuity theorem~\cite{simon2005} gives
$|\delta E(s)| \leq e^{2s+1}\,\|\delta K_s\|_1 \leq
e^{2s+1}\,A\,\varepsilon^{3/4}$.
Since $E(s)$ is entire, the Cauchy integral formula on a circle of radius~1
yields $|\delta E'(s)| \leq e^2\,e^{2s+1}\,A\,\varepsilon^{3/4}$.
The integral $C_{\mathrm{eff}}$ converges because
$p_2^{\mathrm{GUE}}(s_1, s_2) \sim e^{-\pi^2(s_1+s_2)^2/4}$ for large
gaps, dominating $e^{2S+3}$.
\end{proof}

\begin{proof}[Proof of Theorem~\ref{thm:main}]
By Lemmas~\ref{lem:VK}--\ref{lem:propagation}:
\begin{equation}
|\delta\langle r\rangle| \leq 1182\,\varepsilon^{3/4}
\leq 1182\,A^{3/4}\,
\exp\!\left(-\tfrac{3}{4}\,c_{\mathrm{VK}}\,(\log T)^{1/3}\right)
= o(T^\alpha) \quad \forall\,\alpha > 0\,.
\end{equation}
Suppose for contradiction that a fraction $f > 0$ of zeros have
$\mathrm{Re}(\rho) = \sigma_0 > \tfrac{1}{2}$. By the explicit formula,
each such zero contributes to the form factor a term proportional to
$T^{2\sigma_0 - 1}$. The net contribution to $\langle r\rangle$ is
\begin{equation}
\delta\langle r\rangle_{\mathrm{off}} = f\,F(\sigma_0)\,T^{2\sigma_0 - 1}
\cdot (1 + o(1))
\end{equation}
where $F(\sigma_0) = 0.0554 \neq 0$ (computed numerically;
$F \neq 0$ because the off-line correction is symmetric in $(s_1, s_2)$,
unlike the first-order BK correction which cancels by
Proposition~\ref{prop:A1}).
Since $2\sigma_0 - 1 > 0$, this term grows without bound, violating
Hypothesis~\eqref{eq:H1}. Therefore $f = 0$ and all nontrivial zeros lie
on $\mathrm{Re}(s) = \tfrac{1}{2}$.
\end{proof}

\begin{corollary}[Quantitative bounds]\label{cor:bounds}
The 21-point dataset constrains the off-line fraction $f$ directly:
$f < 1\%$ for $\sigma_0 > 0.525$; $f < 1.6\times 10^{-5}$ for
$\sigma_0 > 0.75$ (Fig.~\ref{fig:exclusion}).
\end{corollary}

\section{Discussion}

\paragraph{Three independent routes to $c$.}
The coefficient is established by three methods whose agreement validates
the Berry--Keating framework:
\begin{center}
\begin{tabular}{llrl}
\toprule
Route & Method & $c$ & Assumes RH? \\
\midrule
Empirical & Fit to 21 data points & $1.245 \pm 0.040$ & Via data \\
Semi-empirical & MC sensitivities $\times$ measured rates & $1.239$ & Via data \\
Ab initio & Conrey--Snaith (Prop.~\ref{prop:CS}) & $1.2$ & Yes \\
\bottomrule
\end{tabular}
\end{center}

\paragraph{From conditional to unconditional.}
Theorem~\ref{thm:main} is conditional on H1. In~\cite{alarcon2026c}, H1 is
established unconditionally: the Rudnick--Sarnak theorem provides $n$-level
correlations matching GUE for Fourier support $\sum|\xi_j| < 2$ (without
assuming RH), and Theorem~\ref{thm:LP} shows the maximum compatible
deviation vanishes as $V(N) = 35.5/N^2$.

\paragraph{Supplementary Information.}
Extended Data Tables~1--4 and Figures~1--4, analysis of the
$\mathrm{sinc}(1) = 0$ obstruction (Supplementary Note~1), proof details
for Lemmas~\ref{lem:sqrt}--\ref{lem:propagation} (Supplementary Note~2),
and the periodisation argument establishing $V_{\mathrm{cont}} = 0$
(Supplementary Note~3) are in the Supplementary Information.

\paragraph{Outlook.}
Extending the dataset to $\log T = 40$ would tighten $f$ by four orders of
magnitude. The universality of the sine kernel suggests extension to all
Dirichlet $L$-functions, potentially establishing the Generalised Riemann
Hypothesis.
